% pandoc/LaTeX template for NIH grant pdfs (minimal)
\PassOptionsToPackage{hyphens}{url}

% page geometry (safe default: 0.5in)
\usepackage[margin=0.5in]{geometry}

% fonts via xelatex
\usepackage{fontspec}
\defaultfontfeatures{Ligatures=TeX}
\setmainfont{Arial}
\setsansfont{Arial}

\usepackage{setspace}

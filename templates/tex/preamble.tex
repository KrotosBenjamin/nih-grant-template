% pandoc/LaTeX template for NIH grant pdfs (minimal)
\usepackage{changepage} % provides the adjustwidth environment
\usepackage{setspace}
\setstretch{1.0} % single-spaced body
\usepackage{geometry}

% Helvetica ≈ Arial; satisfies many FOA requirements—verify for your FOA.
% margins set via _quarto.yml geometry
\usepackage[hidelinks]{hyperref}
\usepackage{microtype}

% Make YAML title exactly 11pt
% [B] Section heading block:
\usepackage{titlesec}
\titleformat{\section}{\bfseries\normalsize\filcenter}{}{0pt}{\MakeUppercase}
\titlespacing*{\section}{0pt}{0pt}{0.2ex}
\titlespacing*{\subsection}{0pt}{0.5ex}{0.2ex}
\titleformat{\subsection}{\itshape\normalsize}{}{0pt}{}

% Kill page numbers
\AtBeginDocument{\pagenumbering{gobble}\pagestyle{empty}}

% kill hyperlinks only in bibliography
\usepackage{etoolbox}
\AtBeginEnvironment{CSLReferences}{%
  \let\oldhref\href
  \let\oldurl\url
  \providecommand{\doi}[1]{#1}% some CSL styles emit \doi{...}
  \renewcommand{\href}[2]{#2}% keep visible text, drop link
  \renewcommand{\url}[1]{#1}% print URL as plain text
}
\AtEndEnvironment{CSLReferences}{%
  \let\href\oldhref
  \let\url\oldurl
}
